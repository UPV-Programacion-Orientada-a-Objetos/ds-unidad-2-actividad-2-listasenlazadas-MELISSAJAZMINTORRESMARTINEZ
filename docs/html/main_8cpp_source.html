192/**************************************************************************
193 * @class RotorDeMapeo
194 * @brief Lista circular doblemente enlazada que implementa un rotor de cifrado
195 *
196 * @details
197 * Simula un disco de cifrado similar a una rueda de César. Contiene el
198 * alfabeto A-Z en una lista circular que puede rotarse para cambiar el mapeo.
199 * La posición actual del rotor determina cómo se decodifica cada carácter.
200 **************************************************************************/
201class RotorDeMapeo {
202private:
203 NodoRotor* head;
204 int size;
205
206public:
211 RotorDeMapeo() : head(nullptr), size(0) {
212 // Construir lista circular A..Z
213 NodoRotor* first = nullptr;
214 NodoRotor* prev = nullptr;
215 for (char ch = 'A'; ch <= 'Z'; ++ch) {
216 NodoRotor* n = new NodoRotor(ch);
217 if (!first) first = n;
218 if (prev) {
219 prev->next = n;
220 n->prev = prev;
221 }
222 prev = n;
223 ++size;
224 }
225 // Cerrar la circularidad
226 if (first && prev) {
227 first->prev = prev;
228 prev->next = first;
229 head = first;
230 }
231 }
232
237 ~RotorDeMapeo() {
238 if (!head) return;
239 // Romper circularidad
240 head->prev->next = nullptr;
241 NodoRotor* cur = head;
242 while (cur) {
243 NodoRotor* nx = cur->next;
244 delete cur;
245 cur = nx;
246 }
247 }
248
255 void rotar(int N) {
256 if (!head || size <= 1) return;
257
258 // Calcular desplazamiento efectivo
259 int effective = N % size;
260 if (effective < 0) effective += size;
261
262 // Mover head
263 for (int i = 0; i < effective; ++i) {
264 head = head->next;
265 }
266
267 cout << " -> ROTANDO ROTOR " << (N >= 0 ? "+" : "") << N
268 << " (efectivo: +" << effective << ")" << endl;
269 }

280 char getMapeo(char in) {
281 if (in == ' ') return ' ';
282
283 // Normalizar a mayúscula
284 if (in >= 'a' && in <= 'z')
285 in = char(in - 'a' + 'A');
286
287 if (in < 'A' || in > 'Z')
288 return in;
289
290 // Encontrar posición y devolver carácter mapeado
291 int index = in - 'A';
292 NodoRotor* cur = head;
293 for (int i = 0; i < index; ++i) {
294 cur = cur->next;
295 }
296 return cur->c;
297 }

515void trim(char* s) {
516 // Trim izquierdo
517 int i = 0;
518 while (s[i] && isspace((unsigned char)s[i])) ++i;
519 if (i > 0) memmove(s, s + i, strlen(s + i) + 1);
520
521 // Trim derecho
522 int L = (int)strlen(s);
523 while (L > 0 && isspace((unsigned char)s[L-1]))
524 s[--L] = '\0';
525}

537TramaBase* parseLinea(const char* lineaC) {
538 char buffer[128];
539 strncpy(buffer, lineaC, sizeof(buffer));
540 buffer[sizeof(buffer)-1] = '\0';
541 trim(buffer);
542
543 if (strlen(buffer) == 0) return nullptr;
544
545 // Tokenizar por coma
546 char* token = strtok(buffer, ",");
547 if (!token) return nullptr;
548 trim(token);
549 if (strlen(token) == 0) return nullptr;
550
551 // Trama LOAD
552 if ((token[0] == 'L' || token[0] == 'l') && (token[1] == '\0')) {
553 char* arg = strtok(nullptr, ",");
554 if (!arg) {
555 cout << "Trama L sin argumento." << endl;
556 return nullptr;
557 }
558 trim(arg);
559
560 // Detectar "Space"
561 if ((arg[0] == 'S' || arg[0] == 's') && (strcasecmp(arg, "Space") == 0)) {
562 return new TramaLoad(' ');
563 }
564
565 // Carácter único
566 if (strlen(arg) == 1) {
567 return new TramaLoad(arg[0]);
568 }
569
570 // Tomar primer carácter
571 if (strlen(arg) > 1) {
572 return new TramaLoad(arg[0]);
573 }
574 return nullptr;
575 }
576 // Trama MAP
577 else if ((token[0] == 'M' || token[0] == 'm') && (token[1] == '\0')) {
578 char* arg = strtok(nullptr, ",");
579 if (!arg) {
580 cout << "Trama M sin argumento." << endl;
581 return nullptr;
582 }
583 trim(arg);
584 int n = atoi(arg);
585 return new TramaMap(n);
586 }
587 else {
588 cout << "Tipo de trama desconocido: " << token << endl;
589 return nullptr;
590 }
591}
